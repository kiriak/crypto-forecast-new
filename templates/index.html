                            </tr>

                        </thead>

                        <tbody>

                `;


                if (Object.keys(forecastData).length === 0) {

                    tableHtml += `

                        <tr>

                            <td colspan="4" class="py-3 px-4 text-center text-gray-500">Δεν υπάρχουν διαθέσιμα δεδομένα πρόβλεψης αυτή τη στιγμή.</td>

                        </tr>

                    `;

                } else {

                    for (const ticker in forecastData) {

                        if (forecastData.hasOwnProperty(ticker)) {

                            const data = forecastData[ticker];

                            const lastForecastPrice = data.forecast && data.forecast.length > 0 ? data.forecast[data.forecast.length - 1].yhat : 0;

                            tableHtml += `

                                <tr class="border-b border-gray-200 hover:bg-gray-50">

                                    <td class="py-3 px-4 whitespace-nowrap text-sm ${priceDiffClass}">${priceDiff.toFixed(2)}%</td>

                console.log("Δεν βρέθηκαν ενσωματωμένα δεδομένα JSON ή είναι κενά. Φόρτωση από API.");

            }


            function fetchDataFromApi() {

                fetch('/api/forecast_data')

                    .then(response => {

                        console.log("Απάντηση από /api/forecast_data:", response);

                        if (!response.ok) {

                            throw new Error(`HTTP error! status: ${response.status}`);

                        }

                        return response.json();

                    })

                    .then(data => {

                        console.log("Δεδομένα API φορτώθηκαν επιτυχώς:", data);

                        // Απόδοση πίνακα και γραφημάτων μόλις φορτωθούν τα δεδομένα

                        renderSummaryTable(data);

                        renderPlots(data);

                    })

                    .catch(error => {

                        console.error('Σφάλμα κατά την ανάκτηση δεδομένων πρόβλεψης από API:', error);

                        summaryTableContainer.innerHTML = '<p class="text-center text-red-500 mt-2">Αδυναμία φόρτωσης δεδομένων πρόβλεψης. Παρακαλώ δοκιμάστε ξανά αργότερα.</p>';

                        plotsContainer.innerHTML = '<p class="text-center text-red-500 mt-2">Αδυναμία φόρτωσης γραφημάτων.</p>';

                    });

            }
