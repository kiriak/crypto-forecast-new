            <!-- Το γράφημα θα φορτωθεί εδώ -->

            {% if plot_html %}

    <script>

        document.getElementById('generate-forecast-btn').addEventListener('click', async () => {

            const coinSelector = document.getElementById('coin-selector');

            const selectedCoin = coinSelector.value;

            const plotContainer = document.getElementById('plot-container');

            const errorMessage = document.getElementById('error-message');

            

            // Εμφάνιση μηνύματος φόρτωσης

            plotContainer.innerHTML = '<div class="text-center text-gray-500 text-lg">Παρακαλώ περιμένετε...</div>';

            errorMessage.classList.add('hidden');

            

            try {

                const response = await fetch('/forecast', {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json'

                    },

                    body: JSON.stringify({ coin_symbol: selectedCoin })

                });


                const result = await response.json();


                if (result.error) {

                    plotContainer.innerHTML = `<div class="text-red-500 text-lg p-4">${result.error}</div>`;

                    errorMessage.textContent = result.error;

                    errorMessage.classList.remove('hidden');

                } else {

                    plotContainer.innerHTML = result.plot_html;

                    document.getElementById('last-updated').textContent = new Date().toLocaleString();

                }

            } catch (error) {

                console.error('Σφάλμα κατά την ανάκτηση της πρόβλεψης:', error);

                plotContainer.innerHTML = `<div class="text-red-500 text-lg p-4">Αποτυχία φόρτωσης του γραφήματος. Παρακαλώ δοκιμάστε ξανά.</div>`;

                errorMessage.textContent = 'Αποτυχία φόρτωσης του γραφήματος.';

                errorMessage.classList.remove('hidden');

            }

        });

    </script>
