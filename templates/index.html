    <script>

        document.getElementById('forecast-form').addEventListener('submit', async function(event) {

            event.preventDefault();


            const form = event.target;

            const coinSymbol = form.querySelector('#coin_symbol').value;

            const plotContainer = document.getElementById('plot-container');

            const loadingOverlay = document.getElementById('loading-overlay');

            const submitButton = document.getElementById('submit-btn');


            // Show loading state

            plotContainer.innerHTML = ''; // Clear previous plot

            loadingOverlay.classList.remove('hidden');

            submitButton.disabled = true;


            try {

                const response = await fetch('/forecast', {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json'

                    },

                    body: JSON.stringify({ coin_symbol: coinSymbol })

                });


                if (!response.ok) {

                    throw new Error(`HTTP error! status: ${response.status}`);

                }


                const result = await response.json();


                if (result.error) {

                    plotContainer.innerHTML = `<div class="bg-red-100 border border-red-400 text-red-700 px-4 py-3 rounded-lg relative" role="alert"><span class="block sm:inline">${result.error}</span></div>`;

                } else {

                    plotContainer.innerHTML = result.plot_html;

                }

            } catch (error) {

                console.error('An error occurred during the forecast request:', error);

                plotContainer.innerHTML = `<div class="bg-red-100 border border-red-400 text-red-700 px-4 py-3 rounded-lg relative" role="alert"><span class="block sm:inline">An unexpected error occurred. Please try again.</span></div>`;

            } finally {

                // Hide loading state

                loadingOverlay.classList.add('hidden');

                submitButton.disabled = false;

            }

        });

    </script>
